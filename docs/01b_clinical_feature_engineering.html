Introduction

This script represents Phase 1b of the pipeline. It takes the technically synchronized data from Phase 1a and performs rigorous biological feature engineering, ordinal mapping, and statistical Quality Control. It addresses extreme skewness and class imbalance, generating a finalized curated_clinical_spine.rds and a clinical_data_dictionary.rds for all downstream omics scripts.

Schema & Mapping Enforcement

Analysis Decisions:
  • Single Source of Truth: All variable renaming, exclusion dropping, and datatype coercion is now dictated entirely by an external Excel file (clinical_master_mapping.xlsx).
  • Rename First Rule: Variables are renamed immediately upon import. All subsequent downstream engineering relies exclusively on the ‘Clean_Name’.
  • Global Lowercase: All textual and categorical variables are swept to lowercase to definitively eliminate capitalization artifacts (e.g., ‘Yes’ vs ‘yes’).
What this code does:
  • Loaded the Master Mapping Excel file.
  • Executed a ‘Pre-Flight Check’ to identify raw variables missing from the dictionary.
  • Renamed all variables to their Clean_Name.
  • Dropped all variables tagged with ‘x’.
  • Coerced remaining columns into pure Numerics (n), Categorical Factors (c), or Dates (d).
  • Forced all character columns to lowercase. 
# Initialize the global audit log to track every change made to the dataset
audit_log <- c("Phase 1: Schema & Mapping Audit Log\n")
curated_spine <- copy(raw_spine)

# 1. Load Master Mapping File
if (!file.exists(mapping_path)) stop("FATAL: clinical_master_mapping.xlsx not found.")
mapping_df <- read_excel(mapping_path)

# 2. Pre-Flight Check (Detect Unmapped Variables)
raw_cols <- names(curated_spine)
mapped_cols <- mapping_df$Raw_Name
unmapped <- setdiff(raw_cols, mapped_cols)

if(length(unmapped) > 0) {
  audit_log <- c(audit_log, sprintf("- **WARNING:** %d columns in raw data are missing from the mapping file.", length(unmapped)), setdiff(colnames(raw_spine), mapping_df$Raw_Name))
}

# 3. Rename First, Forget the Past
# Create a named vector: c("Old_Name" = "Clean_Name")
rename_dict <- setNames(mapping_df$Clean_Name, mapping_df$Raw_Name)
setnames(curated_spine, old = names(rename_dict), new = as.character(rename_dict), skip_absent = TRUE)

audit_log <- c(audit_log, sprintf("- Applied Master Naming Dictionary. Renamed %d variables. All subsequent steps now use 'Clean_Name'.", length(rename_dict)))

# 4. The Purge (Drop 'x' variables)
drop_vars <- mapping_df %>% filter(Data_Type == "x") %>% pull(Clean_Name)
actual_drops <- intersect(drop_vars, names(curated_spine))

if(length(actual_drops) > 0) {
  curated_spine[, (actual_drops) := NULL]
  audit_log <- c(audit_log, sprintf("- **Dropped %d variables** designated as 'x' (Excluded) in master mapping.", length(actual_drops)))
}

# 5. Safe Type Coercion
# Extract the target column names based on their flags
num_vars  <- mapping_df %>% filter(Data_Type == "n") %>% pull(Clean_Name) %>% intersect(names(curated_spine))
cat_vars  <- mapping_df %>% filter(Data_Type == "c") %>% pull(Clean_Name) %>% intersect(names(curated_spine))
date_vars <- mapping_df %>% filter(Data_Type == "d") %>% pull(Clean_Name) %>% intersect(names(curated_spine))

# Define helper functions to prevent R from crashing on weird text
safe_numeric <- function(x) { suppressWarnings(as.numeric(gsub(",", ".", trimws(as.character(x))))) }
safe_excel_date <- function(x) { suppressWarnings(as.Date(as.numeric(as.character(x)), origin = "1899-12-30")) }

# Apply the coercions efficiently using data.table syntax
if(length(num_vars) > 0)  curated_spine[, (num_vars)  := lapply(.SD, safe_numeric), .SDcols = num_vars]
if(length(cat_vars) > 0)  curated_spine[, (cat_vars)  := lapply(.SD, as.factor), .SDcols = cat_vars]
if(length(date_vars) > 0) curated_spine[, (date_vars) := lapply(.SD, safe_excel_date), .SDcols = date_vars]

audit_log <- c(audit_log, sprintf("- Enforced Datatypes: Coerced **%d Numeric**, **%d Categorical**, and **%d Date** variables.", length(num_vars), length(cat_vars), length(date_vars)))

# 6. Global Lowercase Sweep
char_cols <- names(curated_spine)[sapply(curated_spine, function (x) is.character(x) || is.factor(x))]
char_cols <- setdiff(char_cols, c("Subject_ID", date_vars))

if (length(char_cols) > 0) {
  curated_spine[, (char_cols) := lapply(.SD, tolower), .SDcols = char_cols]
  audit_log <- c(audit_log, sprintf("- Standardized %d categorical/character variables to lowercase.", length(char_cols)))
}

# Convert categoricals to factors after lowering
if(length(cat_vars) > 0) curated_spine[, (cat_vars) := lapply(.SD, as.factor), .SDcols = cat_vars]

# Print the Phase 1 Log
cat(paste(audit_log, collapse = "\n\n"))

Feature Engineering

What this code does:
  • Engineered cohort groups strictly from Subject_ID prefixes.
  • Calculated standard baseline demographics (Age, Sex, Sample_Age) and temporal deltas between clinical assessments and blood draws.
  • Permanently encoded Subsets (Diffuse/Limited/Sine) as One-Hot binary variables.
  • Extracted individual medications from grouped text strings into discrete binary features. 
audit_log <- c("Phase 2: Feature Engineering Audit Log\n")

# 1. Cohort Group Assignment
curated_spine[, cohort_group := fcase(
  str_detect(Subject_ID, "^Nonact"), GRP_NON_ACTIVE,
  str_detect(Subject_ID, "^Act"),    GRP_ACTIVE,
  str_detect(Subject_ID, "^Ctrl"),   GRP_CONTROL,
  default = NA_character_
)]
curated_spine[, cohort_group := factor(cohort_group, levels = c(GRP_ACTIVE, GRP_NON_ACTIVE, GRP_CONTROL))]

# 2. Demographics (Age and Sample Age)
# Note: Relies on the standard names in your mapping file
if (all(c("Entnahmedatum", "Geburtsdatum") %in% names(curated_spine))) {

if ("Geschlecht" %in% names(curated_spine)) {

# 4. Dynamic Date Deltas (Using the dictionary's 'd' tags)
# Find all columns tagged as dates, excluding the base reference dates
assessment_dates <- setdiff(date_vars, c("Visit Date", "Entnahmedatum", "Einlagerungsdatum", "Geburtsdatum"))

if (length(assessment_dates) > 0 && "Entnahmedatum_Clean" %in% names(curated_spine)) {
  for (d_col in assessment_dates) {
    # Strip procedural text to create a clean prefix for the delta column
    clean_prefix <- gsub(".*Date of assessment for |.*Date of |.*date of ", "", d_col)

}



# 5. Medication Re-engineering (Explosion & One-Hot)
med_cols <- c("Immunosuppressants", "Major vascular", "Steroids", "Other_medication")
actual_med_cols <- intersect(med_cols, names(curated_spine))

if (length(actual_med_cols) > 0) {
  all_drugs <- c()
  
  # Pass 1: Extract universe of unique drugs
  for (m in actual_med_cols) {
    vec <- as.character(curated_spine[[m]])
    vec <- vec[!is.na(vec) & vec != "" & vec != "no" & vec != "none"]
    split_drugs <- unlist(strsplit(vec, "[,;]"))
    clean_drugs <- trimws(split_drugs)
    clean_drugs <- clean_drugs[clean_drugs != "yes" & clean_drugs != "no" & clean_drugs != "" & clean_drugs != "none"]
    all_drugs <- unique(c(all_drugs, clean_drugs))
  }

  # Pass 2: Create binary columns for specific drugs
  for (drug in all_drugs) {
    col_name <- paste0("Med_", gsub("[^A-Za-z0-9]", "_", drug))
    has_drug <- apply(curated_spine[, ..actual_med_cols], 1, function(row) {
      any(grepl(paste0("\\b", regex(drug), "\\b"), row, ignore.case = TRUE))
    })
    curated_spine[, (col_name) := factor(ifelse(has_drug, "yes", "no"), levels = c("no", "yes"))]
  }

  # Pass 3: Overwrite parent columns to clean yes/no
  for (m in actual_med_cols) {
    vec <- as.character(curated_spine[[m]])
    is_yes <- !is.na(vec) & vec != "" & vec != "no" & vec != "none"
    curated_spine[, (m) := factor(ifelse(is_yes, "yes", "no"), levels = c("no", "yes"))]
  }
  
  audit_log <- c(audit_log, sprintf("- Parsed medication strings: Extracted %d specific drugs into one-hot variables and binarized parent classes.", length(all_drugs)))
}

Numeric Spine Generation

What this code does:
  • Generated a parallel curated_spine_numeric by mapping severity rankings (e.g., ‘none’, ‘rare’, ‘moderate’) into mathematical ordinal scales (0-3).
  • Explicitly binarized the Sex and Cohort Group variables.
  • Output a visual tracker to audit exactly which mappings were applied to which variables. 
curated_spine_numeric <- copy(curated_spine)

ordinal_log_df <- data.frame(Variable = character(), Mappings_Applied = character(), stringsAsFactors = FALSE)
scan_cols <- names(curated_spine)[sapply(curated_spine, function(x) is.character(x) || is.factor(x))]

for (col in scan_cols) {
  vec <- tolower(as.character(curated_spine[[col]]))
  changed <- FALSE
  map_applied <- c()
  
  # Check against all hierarchical dictionaries
  for (map_name in names(ordinal_map)) {
    mapping <- ordinal_map[[map_name]]
    if (any(vec %in% names(mapping), na.rm = TRUE)) {
      vec <- unname(mapping[vec]) %>% coalesce(vec)
      changed <- TRUE
      map_applied <- c(map_applied, map_name)
    }
  }
  
  if (changed) {
    curated_spine_numeric[[col]] <- as.numeric(vec)
    ordinal_log_df <- rbind(ordinal_log_df, data.frame(Variable = col, Mappings_Applied = paste(map_applied, collapse = " | ")))
    
  } else if (col == "Sex") {
    curated_spine_numeric[[col]] <- ifelse(curated_spine[[col]] == SEX_FEMALE, 1, 0)
    ordinal_log_df <- rbind(ordinal_log_df, data.frame(Variable = col, Mappings_Applied = "Explicit Binary (Male=0, Female=1)"))
  
  } else if (col == "eustar_group") {
    curated_spine_numeric[[col]] <- ifelse(curated_spine[[col]] == "Active", 1, 0)
    ordinal_log_df <- rbind(ordinal_log_df, data.frame(Variable = col, Mappings_Applied = "Explicit Binary eustar_group (Active=1, Inactive=0)"))
      
  } else if (col == "cohort_group") {
    curated_spine_numeric[[col]] <- case_when(
      curated_spine[[col]] == GRP_CONTROL ~ 0,
      curated_spine[[col]] == GRP_NON_ACTIVE ~ 1,
      curated_spine[[col]] == GRP_ACTIVE ~ 2,
      TRUE ~ NA_real_
    )
    ordinal_log_df <- rbind(ordinal_log_df, data.frame(Variable = col, Mappings_Applied = "Explicit Ordinal (Control=0, Non-Active=1, Active=2)"))
  }
}

Automated QC & Data Dictionary

Analysis Decisions:
  • Continuous variables were evaluated for both skewness and extreme outliers simultaneously.
  • Skewed variables were transformed using the Natural Log: ln(x + c). Variables producing NaNs were flagged and skipped.
  • Categorical variables with a minority class <10% were flagged as IMBALANCED.
  • Character variables were deliberately bypassed from statistical QC.
What this code does:
  • Iterated through every clinical feature.
  • Implemented an accumulating flag system (e.g., ‘SKEWED & EXTREME OUTLIERS’).
  • Unified data taxonomy into a single, clean ‘Class’ column (Continuous, Ordinal, Categorical Binary/Nominal, Excluded).
  • Merged statistical QC results with manual exclusion notes from the Master Mapping file to generate the final definitive Data Dictionary. 
calc_skew <- function(x) {
  x <- x[is.finite(x)]
  n <- length(x)
  if (n < 3) return(NA_real_)
  sum((x - mean(x))^3 / n) / (sd(x)^3)
}

calc_outliers <- function(x) {
  x <- x[is.finite(x)]
  if (length(x) < 5) return(0)
  q25 <- quantile(x, 0.25)
  q75 <- quantile(x, 0.75)
  iqr <- q75 - q25
  upper_bound <- q75 + (qc_config$outlier_iqr_multiplier * iqr)
  lower_bound <- q25 - (qc_config$outlier_iqr_multiplier * iqr)
  sum(x > upper_bound | x < lower_bound)
}

      if (all(val_lower %in% c("none", "rare", "moderate", "extensive"))) {
        plot_vec <- factor(val_lower, levels = c("none", "rare", "moderate", "extensive"))
      } else if (all(val_lower %in% c("never", "previously", "current"))) {
        plot_vec <- factor(val_lower, levels = c("never", "previously", "current"))
      } else if (all(val_lower %in% c("no", "yes"))) {
        plot_vec <- factor(val_lower, levels = c("no", "yes"))
      } else {
        plot_vec <- forcats::fct_infreq(as.character(vec_clean))
      }
      
      props <- prop.table(table(plot_vec)) * 100
      lvl_strs <- paste0(names(props), ": ", sprintf("%.0f%%", props))
      if(length(lvl_strs) > 3) lvl_strs <- c(lvl_strs[1:3], "...")
      
      sub_txt <- sprintf("QC: %s\nDist: %s\nMissing: %s", qc_flag, paste(lvl_strs, collapse=", "), miss_pct)
      
      p <- ggplot(data.frame(x=plot_vec), aes(x)) +
        geom_bar(fill="grey70", color="black", alpha=0.8) +
        theme_project_base(10) +
        labs(title = v, subtitle = sub_txt, x=NULL, y=NULL) +
        theme(plot.subtitle = element_text(color = flag_color, face="bold", size=rel(0.85)),
              plot.title = element_text(size=rel(0.9), face="bold"),
              axis.text.x = element_text(angle=45, hjust=1))
    }
    
    plot_list[[v]] <- p
  }
  
  grid_p <- wrap_plots(plot_list, ncol = 3) +
    plot_annotation(title = sprintf("Clinical Domain: %s", gsub("_", " ", dom)),
                    theme = theme(plot.title = element_text(size=18, face="bold")))
  
  dyn_h <- max(5, ceiling(length(plot_list) / 3) * 3 + 1)
  file_name <- paste0(dom, "_atlas.png")
  file_path <- file.path(atlas_dir, file_name)
  ggsave(file_path, grid_p, width=14, height=dyn_h, dpi=150, bg="white")
}
