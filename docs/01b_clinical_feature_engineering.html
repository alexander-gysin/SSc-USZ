Introduction

This script represents Phase 1b of the pipeline. It takes the technically synchronized data from Phase 1a and performs rigorous biological feature engineering, ordinal mapping, and statistical Quality Control (QA/QC). It addresses extreme skewness and class imbalance, generating a finalized curated_clinical_spine.rds and a clinical_data_dictionary.rds for all downstream omics scripts.

Feature Engineering

Analysis Decisions:
  • Engineered cohort groups strictly from Subject_ID prefixes.
  • Calculated standard baseline demographics (Age, Sex, Sample_Age) and temporal deltas between clinical assessments and blood draws.
  • Permanently encoded Subsets (Diffuse/Limited/Sine) as One-Hot binary variables.
  • Generated a Twin ‘Numeric’ Spine where severity rankings were mapped into mathematical ordinal scales (0-3) for downstream correlation engines.
What this code does:
  • Enforced the strict Schema Dictionary via external text file to coerce datatypes (n=numeric, c=factor) and drop flags (x).
  • Renamed verbose columns via the mapping dictionary.
  • Safely parsed raw Excel date serials into standard R Date objects.
  • Applied One-Hot Encoding to subsets.
  • Generated curated_spine_numeric by tracking Ordinal mappings and coercing factors to numerics, accompanied by a visual report.

Automated QC & Data Dictionary

Analysis Decisions:
  • Continuous variables were evaluated for both skewness and extreme outliers simultaneously.
  • Newly generated Log10 variables were forced through the same QC gauntlet to ensure the transformation successfully normalized the data and didn’t mask unit-entry errors.
  • Categorical variables with a minority class <10% were flagged as IMBALANCED.
  • Character variables were deliberately bypassed from statistical QA/QC.
What this code does:
  • Iterated through every clinical feature routing by data type.
  • Implemented an accumulating flag system (e.g., ‘SKEWED & EXTREME OUTLIERS’).
  • Calculated Skewness and Outliers for both raw and newly transformed log variables, using is.finite() to protect against NaN/Inf crashes.
  • Generated the definitive Data Dictionary object detailing the compound QC status. 
calc_skew <- function(x) {
  x <- x[is.finite(x)]
  n <- length(x)
  if (n < 3) return(NA_real_)
  sum((x - mean(x))^3 / n) / (sd(x)^3)
}

calc_outliers <- function(x) {
  x <- x[is.finite(x)]
  if (length(x) < 5) return(0)
  q25 <- quantile(x, 0.25)
  q75 <- quantile(x, 0.75)
  iqr <- q75 - q25
  upper_bound <- q75 + (qc_config$outlier_iqr_multiplier * iqr)
  lower_bound <- q25 - (qc_config$outlier_iqr_multiplier * iqr)
  sum(x > upper_bound | x < lower_bound)
}

    next
    
  # B. NUMERIC / CONTINUOUS LOGIC
  } else if (is.numeric(vec)) {
    if (length(unique(na.omit(vec))) > 5) {
      skew_val <- calc_skew(vec_patients)
      extreme_outliers <- calc_outliers(vec_patients)
      
      # 1. Accumulate Flags for Raw Variable
      flags <- c()
      if (extreme_outliers > 0) flags <- c(flags, "EXTREME OUTLIERS")
      if (!is.na(skew_val) && abs(skew_val) > qc_config$max_skewness) flags <- c(flags, "SKEWED")
      
      qc_flag <- if (length(flags) > 0) paste(flags, collapse = " & ") else "PASS"
      dist_metric <- sprintf("Skew: %.2f | Outliers: %d", skew_val, extreme_outliers)
      action <- if ("SKEWED" %in% flags) "Log10 Transformed" else if ("EXTREME OUTLIERS" %in% flags) "Flagged (Review for unit errors)" else "None"
      
      # 2. Trigger Log Transformation if Skewed
      if ("SKEWED" %in% flags) {
        log_name <- gsub("_+$", "", paste0("log_", gsub("[^A-Za-z0-9]", "_", v)))
        
        # Dynamically shift negative and zero variables to be strictly positive
        min_val <- min(vec_patients, na.rm = TRUE)
        if (is.finite(min_val) && min_val <= 0) {
          shift_val <- abs(min_val) + 1
          curated_spine[, (log_name) := log10(get(v) + shift_val)]
          curated_spine_numeric[, (log_name) := log10(get(v) + shift_val)]
        } else {
          curated_spine[, (log_name) := log10(get(v))]
          curated_spine_numeric[, (log_name) := log10(get(v))]
        }
        
        # 3. Test the NEW log variable
        log_patients <- curated_spine[[log_name]][!is.na(curated_spine$cohort_group) & curated_spine$cohort_group != GRP_CONTROL]
        log_skew <- calc_skew(log_patients)
        log_outliers <- calc_outliers(log_patients)
        
        log_flags <- c()
        if (log_name %in% qc_config$manual_log_rejects) {
          log_flags <- c(log_flags, "LOG REJECTED")
        } else {
          if (log_outliers > 0) log_flags <- c(log_flags, "EXTREME OUTLIERS")
          if (!is.na(log_skew) && abs(log_skew) > qc_config$max_skewness) log_flags <- c(log_flags, "SKEWED")
        }
        
        log_qc_flag <- if (length(log_flags) > 0) paste(log_flags, collapse = " & ") else "PASS"
        log_action  <- if ("LOG REJECTED" %in% log_flags) "Flagged by User" else "Engineered to fix skew"
        log_dist_metric <- sprintf("Skew: %.2f | Outliers: %d", log_skew, log_outliers)

        qc_plots_skew[[v]] <- local({
          plot_df <- data.frame(Raw = vec_patients, Transformed = log_patients)
          p_raw <- ggplot(plot_df, aes(x = Raw)) + geom_histogram(fill = "#c0392b", color = "white", bins = 30) + theme_project_base(10) + labs(title = v, x = "Raw Data")
          p_log <- ggplot(plot_df, aes(x = Transformed)) + geom_histogram(fill = "#2980b9", color = "white", bins = 30) + theme_project_base(10) + labs(title = log_name, x = "Log10 Data")
          p_raw | p_log
        })

        dict_list[[length(dict_list) + 1]] <- data.frame(Variable = log_name, Class = "Continuous", Missing_Pct = miss_pct, Distribution_Metric = log_dist_metric, QC_Flag = log_qc_flag, Action_Taken = log_action)
      }
      
      dict_list[[length(dict_list) + 1]] <- data.frame(Variable = v, Class = "Continuous", Missing_Pct = miss_pct, Distribution_Metric = dist_metric, QC_Flag = qc_flag, Action_Taken = action)
    } else {
      dict_list[[length(dict_list) + 1]] <- data.frame(Variable = v, Class = "Ordinal / Binary", Missing_Pct = miss_pct, Distribution_Metric = "N/A (<5 Levels)", QC_Flag = "PASS", Action_Taken = "None")
    }
    
  # C. FACTOR / CATEGORICAL LOGIC
  } else if (is.factor(vec) || is.logical(vec)) {
    props <- prop.table(table(vec_patients))
    counts <- table(vec_patients)
    min_prop <- if(length(props) > 0) min(props) else 0
    qc_flag <- "PASS"
    action <- "None"
    
    if (length(props) < 2) {
      qc_flag <- "ZERO VARIANCE"
      action <- "Flagged"
    } else if (min_prop < qc_config$min_minority_pct) {
      qc_flag <- "IMBALANCED"
      action <- "Flagged"
      imb_log_list[[v]] <- data.frame(Variable = v, Level = as.character(names(props)), Count = as.numeric(counts), Percentage = as.numeric(props))
    }
    
    dict_list[[length(dict_list) + 1]] <- data.frame(Variable = v, Class = "Categorical", Missing_Pct = miss_pct, Distribution_Metric = sprintf("Minority: %.1f%%", min_prop * 100), QC_Flag = qc_flag, Action_Taken = action)
  }
}

Clinical Domain Atlas

What this code does:
  • Iterates through the global clinical domain lists to construct a comprehensive visual atlas.
  • Dynamically routes continuous variables to histograms (annotated with Median and IQR) and categorical variables to bar charts.
  • Implements smart categorical sorting: recognized ordinal variables are plotted low-to-high, while nominal variables are sorted by frequency (Pareto principle).
  • Bypasses the gitignored output folder by saving images directly to docs/assets/ and generating raw HTML links for web publishing. 
# Point directly to the web-accessible docs folder
atlas_dir <- here::here("docs", "assets", current_file)
if(!dir.exists(atlas_dir)) dir.create(atlas_dir, recursive = TRUE)

for (dom in names(clinical_domains)) {
  
  vars <- intersect(clinical_domains[[dom]], names(curated_spine))
  vars <- setdiff(vars, c("Subject_ID", "cohort_group"))
  if(length(vars) == 0) next
  
  plot_list <- list()
  
  for (v in vars) {
    vec <- curated_spine[[v]]
    vec_clean <- vec[!is.na(vec) & !is.na(curated_spine$cohort_group) & curated_spine$cohort_group != GRP_CONTROL]
    
    qc_row <- data_dictionary %>% filter(Variable == v)
    qc_flag <- if(nrow(qc_row) > 0) qc_row$QC_Flag[1] else "UNKNOWN"
    miss_pct <- if(nrow(qc_row) > 0) qc_row$Missing_Pct[1] else "UNKNOWN"
    
    flag_color <- case_when(
      qc_flag == "PASS" ~ "#27ae60",
      grepl("OUTLIERS|IMBALANCED|ZERO|REJECTED", qc_flag) ~ "#c0392b",
      grepl("SKEWED", qc_flag) ~ "#2980b9",
      TRUE ~ "black"
    )
    
    if (is.numeric(vec_clean)) {
      med <- median(vec_clean, na.rm=TRUE)
      q25 <- quantile(vec_clean, 0.25, na.rm=TRUE)
      q75 <- quantile(vec_clean, 0.75, na.rm=TRUE)
      sub_txt <- sprintf("QC: %s\nDist: Med %.1f [%.1f - %.1f]\nMissing: %s", qc_flag, med, q25, q75, miss_pct)
      
      p <- ggplot(data.frame(x=vec_clean), aes(x)) +
        geom_histogram(fill="grey70", color="white", bins=30) +
        geom_vline(xintercept=med, color="#2c3e50", linetype="dashed", linewidth=1) +
        theme_project_base(10) +
        labs(title = v, subtitle = sub_txt, x=NULL, y=NULL) +
        theme(plot.subtitle = element_text(color = flag_color, face="bold", size=rel(0.85)),
              plot.title = element_text(size=rel(0.9), face="bold"))
              
    } else {
      val_lower <- tolower(as.character(vec_clean))
      
      # SMART SORTING LOGIC: Ordinal vs Frequency
      if (all(val_lower %in% c("none", "rare", "moderate", "extensive"))) {
        plot_vec <- factor(val_lower, levels = c("none", "rare", "moderate", "extensive"))
      } else if (all(val_lower %in% c("never", "previously", "current"))) {
        plot_vec <- factor(val_lower, levels = c("never", "previously", "current"))
      } else if (all(val_lower %in% c("no", "yes"))) {
        plot_vec <- factor(val_lower, levels = c("no", "yes"))
      } else {
        # Nominal Data: Sort by Frequency (Pareto)
        plot_vec <- forcats::fct_infreq(as.character(vec_clean))
      }
      
      props <- prop.table(table(plot_vec)) * 100
      lvl_strs <- paste0(names(props), ": ", sprintf("%.0f%%", props))
      if(length(lvl_strs) > 3) lvl_strs <- c(lvl_strs[1:3], "...")
      
      sub_txt <- sprintf("QC: %s\nDist: %s\nMissing: %s", qc_flag, paste(lvl_strs, collapse=", "), miss_pct)
      
      p <- ggplot(data.frame(x=plot_vec), aes(x)) +
        geom_bar(fill="grey70", color="black", alpha=0.8) +
        theme_project_base(10) +
        labs(title = v, subtitle = sub_txt, x=NULL, y=NULL) +
        theme(plot.subtitle = element_text(color = flag_color, face="bold", size=rel(0.85)),
              plot.title = element_text(size=rel(0.9), face="bold"),
              axis.text.x = element_text(angle=45, hjust=1))
    }
    
    plot_list[[v]] <- p
  }
  
  grid_p <- wrap_plots(plot_list, ncol = 3) +
    plot_annotation(title = sprintf("Clinical Domain: %s", gsub("_", " ", dom)),
                    theme = theme(plot.title = element_text(size=18, face="bold")))
  
  dyn_h <- max(5, ceiling(length(plot_list) / 3) * 3 + 1)
  
  file_name <- paste0(dom, "_atlas.png")
  file_path <- file.path(atlas_dir, file_name)
  ggsave(file_path, grid_p, width=14, height=dyn_h, dpi=150, bg="white")
  
  cat(sprintf("## %s\n\n", gsub("_", " ", dom)))
  cat(sprintf("<img src='assets/%s/%s' style='width: 100%%;'>\n\n", current_file, file_name))
}
