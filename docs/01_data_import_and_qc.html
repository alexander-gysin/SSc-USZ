# ------------------------------------------------------------------------------
# 1. Process Master Clinical File (Sheet 1 only)
# ------------------------------------------------------------------------------
raw_clin <- read_excel(here::here("data", "Patients__20022025_105625_modified.xlsx"), sheet = 1)
setDT(raw_clin) 

# Force the clinical ID to be a character string for safe merging (Silenced)
raw_clin <- raw_clin[, inCentraxx := as.character(inCentraxx)]

# Strict Filter: Keep ONLY samples explicitly marked for this study
raw_clin <- raw_clin[get(qc_config$selected_col) == 1 | get(qc_config$selected_col) == "1"]

# ------------------------------------------------------------------------------
# 1.5 Fix Excel Numeric Dates
# ------------------------------------------------------------------------------
# Safely target only the date columns that actually exist in the file
valid_date_cols <- intersect(qc_config$date_cols, names(raw_clin))

if(length(valid_date_cols) > 0) {
  # Silenced assignment
  raw_clin <- raw_clin[, (valid_date_cols) := lapply(.SD, function(x) {
    # Suppress warnings in case there is text like "N/A" which will safely become NA
    suppressWarnings(as.Date(as.numeric(x), origin = "1899-12-30"))
  }), .SDcols = valid_date_cols]
}

# ------------------------------------------------------------------------------
# 2. Process Mapping File & Coalesce IDs
# ------------------------------------------------------------------------------
id_map <- read_excel(here::here("data", "Samples for OLink submitted_final_0226.xlsx"))
setDT(id_map)

# Explicitly drop the follow-up sample to enforce cross-sectional design
id_map <- id_map[`Subject ID` != "Act_4_2"]

# Force both ID columns to character (Silenced)
id_map <- id_map[, `Subject-ID (EUSTAR)`      := as.character(`Subject-ID (EUSTAR)`)]
id_map <- id_map[, `Subject-ID (Biobank RUZ)` := as.character(`Subject-ID (Biobank RUZ)`)]

# COALESCE: Create a master 'Merge_ID' (Silenced)
id_map <- id_map[, Merge_ID := fifelse(
  is.na(`Subject-ID (EUSTAR)`) | `Subject-ID (EUSTAR)` == "", 
  `Subject-ID (Biobank RUZ)`, 
  `Subject-ID (EUSTAR)`
)]

# Drop rows from the mapping file that have no ID in either column
id_map <- id_map[!is.na(Merge_ID) & Merge_ID != ""]

# ------------------------------------------------------------------------------
# 3. Merge and Structure
# ------------------------------------------------------------------------------
# Inner join: keeps only participants present in BOTH the clinical and mapping files
master_spine <- merge(
  raw_clin, 
  id_map, 
  by.x = "inCentraxx", 
  by.y = "Merge_ID", 
  all = FALSE
)

# ------------------------------------------------------------------------------
# 4. Group Assignment
# ------------------------------------------------------------------------------
master_spine <- setnames(master_spine, old = "Subject ID", new = "Subject_ID", skip_absent = TRUE)

# Assign Cohort Group based on the Subject_ID prefix (Silenced)
master_spine <- master_spine[, cohort_group := fcase(
  str_detect(Subject_ID, "^Nonact"), GRP_NON_ACTIVE,
  str_detect(Subject_ID, "^Act"),    GRP_ACTIVE,
  str_detect(Subject_ID, "^Ctrl"),   GRP_CONTROL,
  default = NA_character_
)]

# Drop any rows that couldn't be classified (failsafe)
master_spine <- master_spine[!is.na(cohort_group)]

# Reorder columns to make Subject_ID the true Primary Key at the front
master_spine <- setcolorder(master_spine, c("Subject_ID", "cohort_group", "inCentraxx"))

cat("Running Missingness QC and Imputation...\n")
Running Missingness QC and Imputation...
# 1. Quality Control (Missingness Rules from qc_config)
miss_prots <- rowMeans(is.na(olink_mat))
miss_samps <- colMeans(is.na(olink_mat))

valid_prots <- miss_prots < qc_config$max_missing_pct_prot
valid_samps <- miss_samps < qc_config$max_missing_pct_pat

# Extract the names of the failures before we drop them
failed_prots <- rownames(olink_mat)[!valid_prots]
failed_samps <- colnames(olink_mat)[!valid_samps]

# --- Print Protein QC Results ---
if(length(failed_prots) > 0) {
  cat(sprintf("! Dropped %d Proteins (>= %s%% missing):\n  %s\n", 
              length(failed_prots), qc_config$max_missing_pct_prot * 100, paste(failed_prots, collapse = ", ")))
} else {
  cat(sprintf("✓ All proteins passed QC (< %s%% missing).\n", qc_config$max_missing_pct_prot * 100))
}
✓ All proteins passed QC (< 5% missing).
# --- Print Patient QC Results ---
if(length(failed_samps) > 0) {

} else {
  cat(sprintf("✓ All patients passed QC (< %s%% missing).\n", qc_config$max_missing_pct_pat * 100))

✓ All patients passed QC (< 5% missing).
