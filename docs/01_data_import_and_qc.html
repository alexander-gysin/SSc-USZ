What this code does:

  • Set the clinical variable completeness threshold (variables must be 80% complete in patients to survive).

  • Parameterized all raw data file names to avoid hardcoding deep in the script.

Clinical Data Cleanup

Analysis Decisions:
  • Dropped the follow-up sample ‘Act_4_2’ to strictly enforce a cross-sectional study design.
  • Excluded patients who were missing from either the CentraXX output or the Biobank sample manifest.
What this code does:
  • Imported raw clinical and sample mapping Excel files and resolved ID discrepancies.
  • Engineered core analytical features including Age, Sample_Age, cohort_group, and Sex.
  • Calculated time deltas (in years) between clinical assessment dates and the blood draw date.
  • Pruned clinical variables according to the defined 80% completeness filter.
  • Enforced strict data typing and dropped manually excluded variables using the clinical_dictionary.
  • Dynamically dropped procedural meta-variables based on parameterized config patterns.
  • Standardized factor levels: converted ‘UN’ and ‘Unknown’ to NA.
  • Finalized downstream features (renamed long columns, fixed illogical CRP values, locked factor levels). 
audit_log <- c("Importing Clinical Data and ID Maps...")


# 1. Process Master Clinical File (Sheet 1 only) -------------------------------
raw_clin <- read_excel(here::here("data", file_paths$clinical_raw), sheet = 1)
setDT(raw_clin) 

# Force the clinical ID to be a character string for safe merging
raw_clin <- raw_clin[, inCentraxx := as.character(inCentraxx)]

# Strict Filter: Keep ONLY samples explicitly marked for this study
raw_clin <- raw_clin[get(qc_config$selected_col) == 1 | get(qc_config$selected_col) == "1"]


# 1.5 Fix Excel Numeric Dates (DIRECT PARSING) ---------------------------------
valid_date_cols <- intersect(qc_config$date_cols, names(raw_clin))

if(length(valid_date_cols) > 0) {
  raw_clin <- raw_clin[, (valid_date_cols) := lapply(.SD, function(x) {
    suppressWarnings(as.Date(as.numeric(x), origin = "1899-12-30"))
  }), .SDcols = valid_date_cols]
}


# 2. Process Mapping File & Coalesce IDs ---------------------------------------
id_map <- read_excel(here::here("data", file_paths$id_map))
setDT(id_map)

# Explicitly drop the follow-up sample to enforce cross-sectional design
id_map <- id_map[`Subject ID` != "Act_4_2"]

id_map <- id_map[, `Subject-ID (EUSTAR)`      := as.character(`Subject-ID (EUSTAR)`)]
id_map <- id_map[, `Subject-ID (Biobank RUZ)` := as.character(`Subject-ID (Biobank RUZ)`)]

# COALESCE: Create a master 'Merge_ID'
id_map <- id_map[, Merge_ID := fifelse(
  is.na(`Subject-ID (EUSTAR)`) | `Subject-ID (EUSTAR)` == "", 
  `Subject-ID (Biobank RUZ)`, 
  `Subject-ID (EUSTAR)`
)]

id_map <- id_map[!is.na(Merge_ID) & Merge_ID != ""]


# 3. Merge and Structure -------------------------------------------------------
master_spine <- merge(
  raw_clin, 
  id_map, 
  by.x = "inCentraxx", 
  by.y = "Merge_ID", 
  all = FALSE
)

# SANITY CHECK: Ensure the merge actually yielded data
stopifnot("FATAL: Inner join resulted in 0 rows. Check ID formatting." = nrow(master_spine) > 0)
audit_log <- c(audit_log, "- Sanity Check: Inner join successful.")

setnames(master_spine, "Right atrium area (cm²)", "Right atrium area (cm2)", skip_absent = TRUE)
setnames(master_spine, "Right ventricular area (cm²)", "Right ventricular area (cm2)", skip_absent = TRUE)


# 4. Feature Engineering (Groups, Dates, Age, Sex) -----------------------------
master_spine <- setnames(master_spine, old = "Subject ID", new = "Subject_ID", skip_absent = TRUE)

master_spine <- master_spine[, cohort_group := fcase(
  str_detect(Subject_ID, "^Nonact"), GRP_NON_ACTIVE,
  str_detect(Subject_ID, "^Act"),    GRP_ACTIVE,
  str_detect(Subject_ID, "^Ctrl"),   GRP_CONTROL,
  default = NA_character_
)]

master_spine <- master_spine[, Entnahmedatum_Clean := as.Date(Entnahmedatum)]

master_spine <- master_spine[, Birth_Year := as.numeric(trimws(Geburtsdatum))]
master_spine <- master_spine[, Age := year(Entnahmedatum_Clean) - Birth_Year]
master_spine <- master_spine[, Sample_Age := as.numeric(difftime(qc_config$olink_date, Entnahmedatum_Clean, units = "days")) / 365.25]

master_spine <- master_spine[, Sex := fcase(
  Geschlecht == "weiblich", SEX_FEMALE,
  Geschlecht == "männlich", SEX_MALE,
  default = NA_character_
)]


# 4.5 Feature Engineering (Clinical Date Deltas) -------------------------------
assessment_dates <- setdiff(qc_config$date_cols, c("Visit Date", "Entnahmedatum", "Einlagerungsdatum"))

for (d_col in assessment_dates) {
  if (d_col %in% names(master_spine)) {
    clean_prefix <- gsub(".*Date of assessment for |.*Date of ", "", d_col)
    clean_name <- paste0("Delta_", gsub("[^A-Za-z0-9]", "_", clean_prefix), "_Years")
    master_spine[, (clean_name) := as.numeric(difftime(get(d_col), Entnahmedatum_Clean, units = "days")) / 365.25]
  }
}


# 5. Completeness Pruning & Dictionary Enforcement -----------------------------
audit_log <- c(audit_log, "\nFiltering by completeness and applying TSV Data Dictionary...")

protected_vars <- c("Subject_ID", "inCentraxx", "cohort_group", "Age", "Sex", "Sample_Age", qc_config$date_cols)

completeness <- master_spine[cohort_group %in% c(GRP_ACTIVE, GRP_NON_ACTIVE), 
                             lapply(.SD, function(x) sum(!is.na(x)) / .N)]

cols_to_keep <- names(completeness)[completeness >= qc_config$min_completeness_clin_var | names(completeness) %in% protected_vars]

n_dropped <- ncol(master_spine) - length(cols_to_keep)
master_spine <- master_spine[, ..cols_to_keep]
audit_log <- c(audit_log, sprintf("- Pruning: Dropped %d columns for having < %s%% completeness in patients.", 
            n_dropped, qc_config$min_completeness_clin_var * 100))

dict_path <- here::here("data", file_paths$clinical_dict)
if (file.exists(dict_path)) {
  data_dict <- readr::read_tsv(dict_path, show_col_types = FALSE)
  
  vars_n <- intersect(data_dict %>% filter(Type == "n") %>% pull(Variable_Name), names(master_spine))
  vars_c <- intersect(data_dict %>% filter(Type == "c") %>% pull(Variable_Name), names(master_spine))
  vars_x <- intersect(data_dict %>% filter(Type == "x") %>% pull(Variable_Name), names(master_spine))
  
  if (length(vars_x) > 0) master_spine[, (vars_x) := NULL]
  
  safe_numeric <- function(x) {
    if (is.numeric(x)) return(x)
    clean_x <- gsub(",", ".", trimws(as.character(x)))
    suppressWarnings(as.numeric(clean_x))
  }
  if (length(vars_n) > 0) master_spine[, (vars_n) := lapply(.SD, safe_numeric), .SDcols = vars_n]
  if (length(vars_c) > 0) master_spine[, (vars_c) := lapply(.SD, function(x) as.factor(as.character(x))), .SDcols = vars_c]
  
  audit_log <- c(audit_log, sprintf("- Enforcement: Cleaned %d Numeric and %d Categorical variables based on dictionary.", length(vars_n), length(vars_c)))
  audit_log <- c(audit_log, sprintf("- Exclusion: Dropped %d manually blacklisted variables.", length(vars_x)))
} else {
  audit_log <- c(audit_log, "! WARNING: 'clinical_dictionary.txt' not found in data/. Skipping explicit type enforcement.")
}


# 5.5 Meta-Variable Pruning (Dropping Measurement Indicators) ------------------
audit_log <- c(audit_log, "\nPruning procedural meta-variables (measurement indicators)...")

regex_pattern <- paste(tolower(qc_config$meta_drop_patterns), collapse = "|")
cols_to_drop <- names(master_spine)[str_detect(tolower(names(master_spine)), regex_pattern)]

if (length(cols_to_drop) > 0) {
  master_spine <- master_spine %>% select(-all_of(cols_to_drop))
  audit_log <- c(audit_log, sprintf("- Dropped %d procedural variables to prevent analytical confounding.", length(cols_to_drop)))
}


# 6. Factor Standardization (Binarization & Cleaning) --------------------------
audit_log <- c(audit_log, "\nStandardizing factor variables (Yes/No -> 1/0, UN/Unknown -> NA)...")

master_spine <- master_spine %>%
  mutate(across(where(is.factor), ~ {
    char_vec <- as.character(.x)
    lower_vec <- tolower(char_vec)
    
    char_vec[lower_vec %in% c("un", "unknown", "not done")] <- NA
    char_vec[lower_vec == "yes"] <- "1"
    char_vec[lower_vec == "no"] <- "0"
    
    as.factor(char_vec)
  }))
setDT(master_spine)


# 7. Final Downstream Feature Engineering --------------------------------------
audit_log <- c(audit_log, "\nApplying final clinical feature engineering (Renaming, Factor levels)...")

setnames(master_spine, 
         old = as.character(clinical_rename_map), 
         new = names(clinical_rename_map), 
         skip_absent = TRUE)

setnames(master_spine, 
         old = c("OR serum creatinine (micromoles/L)", "CRP (mg/dl)"), 
         new = c("Creatinine", "CRP"), 
         skip_absent = TRUE)

if("CRP" %in% names(master_spine)) {
  master_spine[, CRP := as.numeric(gsub(",", ".", as.character(CRP)))]
  master_spine[CRP <= 0, CRP := NA_real_]
}

if("Creatinine" %in% names(master_spine)) {
  master_spine[, Creatinine := as.numeric(gsub(",", ".", as.character(Creatinine)))]
}

master_spine[, cohort_group := factor(cohort_group, levels = c(GRP_ACTIVE, GRP_NON_ACTIVE, GRP_CONTROL))]

core_cols <- intersect(c("Subject_ID", "cohort_group", "Age", "Sex", "Sample_Age"), names(master_spine))
setcolorder(master_spine, core_cols)

audit_log <- c(audit_log, sprintf("\n✓ Clinical Spine Ready: %d rows x %d columns.", nrow(master_spine), ncol(master_spine)))

# Print consolidated output to prevent HTML fragmentation
cat(paste(audit_log, collapse = "\n"))

OLINK QC, Imputation & Sync

Analysis Decisions:
  • Dropped proteins and samples exceeding the 5% missingness threshold (max_missing_pct_prot and max_missing_pct_pat) to ensure robust downstream imputation.
What this code does:
  • Proteins and Samples exceeding the previously defined missingness threshold were entirely dropped.
  • Remaining NA values were imputed using a row-wise (protein-wise) median approach.
  • The master_spine clinical dataset and the Olink matrix were cross-synchronized to guarantee identical sample inclusion and ordering.

audit_log_qc <- c("Running Missingness QC and Imputation...")

miss_prots <- rowMeans(is.na(olink_mat))
miss_samps <- colMeans(is.na(olink_mat))

valid_prots <- miss_prots < qc_config$max_missing_pct_prot
valid_samps <- miss_samps < qc_config$max_missing_pct_pat

failed_prots <- rownames(olink_mat)[!valid_prots]
failed_samps <- colnames(olink_mat)[!valid_samps]

if(length(failed_prots) > 0) {
  audit_log_qc <- c(audit_log_qc, sprintf("! Dropped %d Proteins (>= %s%% missing):\n  %s", 
              length(failed_prots), qc_config$max_missing_pct_prot * 100, paste(failed_prots, collapse = ", ")))
} else {
  audit_log_qc <- c(audit_log_qc, sprintf("✓ All proteins passed QC (< %s%% missing).", qc_config$max_missing_pct_prot * 100))
}

if(length(failed_samps) > 0) {
  audit_log_qc <- c(audit_log_qc, sprintf("! Dropped %d Patients (>= %s%% missing):\n  %s", 
              length(failed_samps), qc_config$max_missing_pct_pat * 100, paste(failed_samps, collapse = ", ")))
} else {
  audit_log_qc <- c(audit_log_qc, sprintf("✓ All patients passed QC (< %s%% missing).", qc_config$max_missing_pct_pat * 100))
}

qc_mat <- olink_mat[valid_prots, valid_samps, drop = FALSE]
audit_log_qc <- c(audit_log_qc, sprintf("\n- Post-QC Matrix: %d Proteins | %d Samples", nrow(qc_mat), ncol(qc_mat)))

row_meds <- matrixStats::rowMedians(qc_mat, na.rm = TRUE)
