Introduction

This script represents Phase 1a of the pipeline. It strictly handles the ingestion of raw data files, resolution of patient identifiers, and technical quality control (missingness filtering and imputation). No biological feature engineering or statistical transformations are performed here.

Setup

What this code does:
  • Loaded required utility packages (tidyverse, data.table, readxl, matrixStats).
  • Established dynamic output directory paths to safely save synchronized data.
  • Configured default knitr chunk options to suppress unnecessary warnings.

Configuration

What this code does:
  • Defined strict quality control missingness thresholds for proteomics filtering (5% max missingness).
  • Set the clinical variable completeness threshold (variables must be 80% complete to survive).
  • Listed protected raw variables to prevent the missingness filter from accidentally dropping foundational columns needed for downstream engineering in Script 01b.
  • Parameterized all raw data file names to avoid hardcoding deep in the script. 
# 1. Global File Paths ---------------------------------------------------------
file_paths <- list(
  clinical_raw  = "clinical_data.xlsx",
  id_map        = "Samples for OLink submitted_final_0226.xlsx",
  olink_raw     = "pre_quant.csv"
)

# 2. Global QC Thresholds & Protected Columns ----------------------------------
qc_config <- list(
  max_missing_pct_prot = 0.05,
  max_missing_pct_pat  = 0.05,
  selected_col         = "SELECTED", 
  
  max_missingness_clin_var = 0.20, 
  
  # Protect these from the 20% NA filter because Script 01b needs them to engineer features
  protected_raw_vars = c(
    "Subject_ID", "inCentraxx", "Geburtsdatum", "Geschlecht", "Entnahmedatum", 
    "Visit Date", "Einlagerungsdatum"
  )
)

Clinical Data

Analysis Decisions:
  • Dropped the follow-up sample ‘Act_4_2’ to strictly enforce a cross-sectional study design.
  • Excluded patients who were missing from either the CentraXX output or the Biobank sample manifest.
  • Converted explicit missing-data strings (‘UN’, ‘Unknown’, ‘Not done’) to true NAs to prevent phantom data from surviving the completeness filter.
  • Dropped any raw clinical column with >20% missingness in SSc patients (excluding controls to prevent artificial NA inflation). Protected foundational variables were retained.
What this code does:
  • Imported raw clinical and sample mapping Excel files and coalesced IDs.
  • Standardized missing value strings into true NAs.
  • Calculated missingness exclusively on patient samples (ignoring healthy controls) and applied the 20% missingness filter.
  • Generated a scrollable HTML audit table detailing exactly which columns were dropped and their missingness percentages. 
audit_log <- c("### Clinical Import Log\n")

# 1. Process Master Clinical File ----------------------------------------------
raw_clin <- read_excel(here::here("data", file_paths$clinical_raw), sheet = 1)
setDT(raw_clin) 

# Force the clinical ID to be a character string for safe merging
raw_clin <- raw_clin[, inCentraxx := as.character(inCentraxx)]

# Strict Filter: Keep ONLY samples explicitly marked for this study
raw_clin <- raw_clin[get(qc_config$selected_col) == 1 | get(qc_config$selected_col) == "1"]

# 2. Process Mapping File & Coalesce IDs ---------------------------------------
id_map <- read_excel(here::here("data", file_paths$id_map))
setDT(id_map)

# Explicitly drop the follow-up sample to enforce cross-sectional design
id_map <- id_map[`Subject ID` != "Act_4_2"]
id_map <- id_map[, `Subject-ID (EUSTAR)`      := as.character(`Subject-ID (EUSTAR)`)]
id_map <- id_map[, `Subject-ID (Biobank RUZ)` := as.character(`Subject-ID (Biobank RUZ)`)]

# COALESCE: Create a master 'Merge_ID'
id_map <- id_map[, Merge_ID := fifelse(
  is.na(`Subject-ID (EUSTAR)`) | `Subject-ID (EUSTAR)` == "", 
  `Subject-ID (Biobank RUZ)`, 
  `Subject-ID (EUSTAR)`
)]

id_map <- id_map[!is.na(Merge_ID) & Merge_ID != ""]

# 3. Merge and Structure -------------------------------------------------------
raw_spine <- merge(
  raw_clin, 
  id_map, 
  by.x = "inCentraxx", 
  by.y = "Merge_ID", 
  all = FALSE
)

# Rename the final subject ID column for global consistency
raw_spine <- setnames(raw_spine, old = "Subject ID", new = "Subject_ID", skip_absent = TRUE)

audit_log <- c(audit_log, sprintf("- Inner join successful. Matched %d unique patients.", nrow(raw_spine)))

# 3.5 Standardize Missing Values ("Phantom Data" Sweep) ------------------------
# Convert case-insensitive "un", "unknown", and "not done" to true NA 
raw_spine <- raw_spine %>%
  mutate(across(where(is.character), ~ if_else(tolower(.x) %in% c("un", "unknown", "not done"), NA_character_, .x)))
setDT(raw_spine) # Ensure it stays a high-performance data.table

audit_log <- c(audit_log, "- Standardized phantom missing data ('UN', 'Unknown') to true NA.")

# 4. Completeness Pruning (The Technical NA Filter) ----------------------------
# CRITICAL: Exclude Healthy Controls from the calculation so their natural NAs don't inflate the missingness!
missingness <- raw_spine[!str_detect(Subject_ID, "^Ctrl"), lapply(.SD, function(x) sum(is.na(x)) / .N)]

# Convert 1-row data.table to a named numeric vector for safe logical filtering
miss_vec <- unlist(missingness)

cols_to_keep <- names(miss_vec)[miss_vec <= qc_config$max_missingness_clin_var | names(miss_vec) %in% qc_config$protected_raw_vars]
cols_to_drop <- setdiff(names(miss_vec), cols_to_keep)

n_dropped <- length(cols_to_drop)
raw_spine <- raw_spine[, ..cols_to_keep]

audit_log <- c(audit_log, sprintf("- Pruning: Dropped %d columns for having > %s%% missingness in the SSc cohort.", 
            n_dropped, qc_config$max_missingness_clin_var * 100))
audit_log <- c(audit_log, sprintf("- Clinical Spine Dimensions: %d rows x %d columns.", nrow(raw_spine), ncol(raw_spine)))

# 5. Build the Dropped Variables Audit Table -----------------------------------
dropped_html <- ""
if (n_dropped > 0) {
  dropped_df <- data.frame(
    Variable = cols_to_drop,
    Missingness = miss_vec[cols_to_drop]
  ) %>% 
    arrange(desc(Missingness)) %>% 
    mutate(Missingness = sprintf("%.1f%%", Missingness * 100))
  
  dropped_tbl <- dropped_df %>%
    kableExtra::kbl(format = "html", row.names = FALSE, 
                    caption = sprintf("<b style='color:#c0392b;'>Excluded Clinical Variables (Missingness > %s%%)</b>", qc_config$max_missingness_clin_var * 100)) %>%
    kableExtra::kable_styling(bootstrap_options = c("striped", "hover", "condensed"), full_width = TRUE, position = "left") %>%
    kableExtra::scroll_box(width = "100%", height = "600px")
  
  if (interactive()) {
    print(dropped_tbl)
  } else {
    dropped_html <- as.character(dropped_tbl)
  }
}

# Cleanup Memory
rm(raw_clin, id_map, missingness, miss_vec, cols_to_keep, cols_to_drop)
if(exists("dropped_df")) rm(dropped_df)
invisible(gc())

# Print HTML output as a single block to prevent code fragmentation
cat(paste(audit_log, collapse = "\n\n"))

Proteomics Data

Analysis Decisions:
  • Dropped proteins and samples exceeding the 5% missingness threshold to ensure robust downstream imputation.
  • NA values were imputed using a row-wise (protein-wise) median approach.
What this code does:
  • Pivoted the Olink data to a wide matrix.
  • Executed missingness filtering and median imputation.
  • Cross-synchronized the clinical spine and the Olink matrix to guarantee identical sample inclusion and ordering. 
audit_log_olink <- c("Proteomics QC Log\n")

# 1. Import and Pivot ----------------------------------------------------------
raw_olink <- fread(here::here("data", file_paths$olink_raw))
raw_olink <- raw_olink[SampleID %in% raw_spine$Subject_ID]

wide_olink <- dcast(raw_olink, Assay ~ SampleID, value.var = "NPX")
olink_mat <- as.matrix(wide_olink[, -1, with = FALSE])
rownames(olink_mat) <- wide_olink$Assay

audit_log_olink <- c(audit_log_olink, sprintf("- Initial Matrix: %d Proteins | %d Samples", nrow(olink_mat), ncol(olink_mat)))

# 2. Missingness Filtering -----------------------------------------------------
miss_prots <- rowMeans(is.na(olink_mat))
miss_samps <- colMeans(is.na(olink_mat))

valid_prots <- miss_prots < qc_config$max_missing_pct_prot
valid_samps <- miss_samps < qc_config$max_missing_pct_pat

failed_prots <- rownames(olink_mat)[!valid_prots]
failed_samps <- colnames(olink_mat)[!valid_samps]

if(length(failed_prots) > 0) {
  audit_log_olink <- c(audit_log_olink, sprintf("- Dropped %d Proteins (>= %s%% missing).", length(failed_prots), qc_config$max_missing_pct_prot * 100))
